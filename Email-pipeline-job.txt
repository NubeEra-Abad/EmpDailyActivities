pipeline {
    agent any
    stages {
        stage('Git-Checkout') {
            steps {
                git credentialsId: 'git', url: 'https://github.com/vishalkhillare/calculator-unit-test-example-java.git'
                echo "Checkout successful"
            }
        }
        
        stage('Compile') {
            steps {
                sh 'mvn compile'
                echo "Compile successful"
            }
        }
        
        stage('Build') {
            steps {
                sh 'mvn clean'
                sh 'mvn package'
                echo "Build successful"
            }
        }
        
        stage('Test') {
            steps {
                sh 'mvn test'
                echo "Test successful"
            }
        }
    }
    
    post {
        always {
            emailext body: "Pipeline ${currentBuild.currentResult}: Job '${env.JOB_NAME} [${env.BUILD_NUMBER}]'\n\nCheck the console output at $BUILD_URL to view the results.",
                recipientProviders: [[$class: 'DevelopersRecipientProvider']],
                subject: "Pipeline ${currentBuild.currentResult}: Job '${env.JOB_NAME} [${env.BUILD_NUMBER}]'",
                to: "tahershaikh755@gmail.com",
                attachLog: true
        }
    }
}
